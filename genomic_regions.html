# Load txdb
txdb <- TxDb.Hsapiens.UCSC.hg38.knownGene

canonical_chromosomes <- paste0("chr", c(1:22, "X", "Y"))
txdb <- keepSeqlevels(txdb, canonical_chromosomes, pruning.mode = "coarse")

# Define and calculate promoter regions
promoter <- getPromoters(TxDb=txdb, upstream=3000, downstream=3000)
promoter_width <- as.numeric(sum(sum(width(reduce(promoter)))))
print(paste("Promoter width:", promoter_width))
## [1] "Promoter width: 185415057"
promoter1kb <- getPromoters(TxDb=txdb, upstream=2000, downstream=2000)
promoter1kb_width <- as.numeric(sum(sum(width(reduce(promoter1kb)))))
print(paste("Promoter 1-2kb width:", promoter1kb_width))
## [1] "Promoter 1-2kb width: 124670769"
promoter2kb <- getPromoters(TxDb=txdb, upstream=1000, downstream=1000)
promoter2kb_width <- as.numeric(sum(sum(width(reduce(promoter2kb)))))
print(paste("Promoter 2-3kb width:", promoter2kb_width))
## [1] "Promoter 2-3kb width: 62996463"
# Calculate 3'UTRs and 5'UTRs
threeUTR <- threeUTRsByTranscript(txdb)
threeUTR_width <- as.numeric(sum(sum(width(reduce(threeUTR)))))
print(paste("3'UTR width:", threeUTR_width))
## [1] "3'UTR width: 111623040"
fiveUTR <- fiveUTRsByTranscript(txdb)
fiveUTR_width <- as.numeric(sum(sum(width(reduce(fiveUTR)))))
print(paste("5'UTR width:", fiveUTR_width))
## [1] "5'UTR width: 21950218"
# Calculate exons and first exons
exons <- exonsBy(txdb, by="tx")
exons <- unlist(exons, use.names=FALSE)
exons_width <- as.numeric(sum(sum(width(reduce(exons)))))
print(paste("Exons width:", exons_width))
## [1] "Exons width: 160291973"
first_exons <- heads(exonsBy(txdb, by="tx"), n=1)
first_exons <- unlist(first_exons, use.names=FALSE)
first_exons_width <- as.numeric(sum(sum(width(reduce(first_exons)))))
print(paste("First exons width:", first_exons_width))
## [1] "First exons width: 53345483"
# Calculate introns and first introns
introns <- intronsByTranscript(txdb)
introns <- unlist(introns, use.names=FALSE)
introns_width <- as.numeric(sum(sum(width(reduce(introns)))))
print(paste("Introns width:", introns_width))
## [1] "Introns width: 1830168913"
first_introns <- heads(intronsByTranscript(txdb), n=1)
first_introns <- unlist(first_introns, use.names=FALSE)
first_introns_width <- as.numeric(sum(sum(width(reduce(first_introns)))))
print(paste("First introns width:", first_introns_width))
## [1] "First introns width: 811694274"
# Calculate intergenic regions
genic <- genes(txdb)
##   27 genes were dropped because they have exons located on both strands
##   of the same reference sequence or on more than one reference sequence,
##   so cannot be represented by a single genomic range.
##   Use 'single.strand.genes.only=FALSE' to get all the genes in a
##   GRangesList object, or use suppressMessages() to suppress this message.
genic <- reduce(genic, ignore.strand=T)
intergenic <- gaps(genic)
intergenic <- intergenic[strand(intergenic) == "*"]
intergenic_width <- as.numeric(sum(sum(width(reduce(intergenic)))))
print(paste("Intergenic width:", intergenic_width))
## [1] "Intergenic width: 1404300076"
# Calculate 3kb downstream regions
downstream <- flank(genes(txdb), width=300, start=FALSE, both=FALSE)
##   27 genes were dropped because they have exons located on both strands
##   of the same reference sequence or on more than one reference sequence,
##   so cannot be represented by a single genomic range.
##   Use 'single.strand.genes.only=FALSE' to get all the genes in a
##   GRangesList object, or use suppressMessages() to suppress this message.
downstream_width <- as.numeric(sum(sum(width(reduce(downstream)))))
print(paste("Downstream 300bp width:", downstream_width))
## [1] "Downstream 300bp width: 9705635"
# Check for NA values
if (is.na(promoter_width) || is.na(promoter1kb_width) || is.na(promoter2kb_width) || is.na(threeUTR_width) ||
    is.na(fiveUTR_width) || is.na(exons_width) || is.na(first_exons_width) || is.na(introns_width) ||
    is.na(first_introns_width) || is.na(intergenic_width) || is.na(downstream_width)) {
  stop("One or more width calculations resulted in NA.")
}

# Summarize all widths
total_width <- promoter_width + promoter1kb_width + promoter2kb_width + threeUTR_width + 
               fiveUTR_width + exons_width + first_exons_width + introns_width + 
               first_introns_width + intergenic_width + downstream_width
print(paste("Total width:", total_width))
## [1] "Total width: 4776161901"
# Convert widths to percentages
promoter_percent <- (promoter_width / total_width) * 100
promoter1kb_percent <- (promoter1kb_width / total_width) * 100
promoter2kb_percent <- (promoter2kb_width / total_width) * 100
threeUTR_percent <- (threeUTR_width / total_width) * 100
fiveUTR_percent <- (fiveUTR_width / total_width) * 100
exons_percent <- (exons_width / total_width) * 100
first_exons_percent <- (first_exons_width / total_width) * 100
introns_percent <- (introns_width / total_width) * 100
first_introns_percent <- (first_introns_width / total_width) * 100
intergenic_percent <- (intergenic_width / total_width) * 100
downstream_percent <- (downstream_width / total_width) * 100

# Define the data for the pie chart
labels <- c("Promoter (<=1kb)", "Promoter (1-2kb)", "Promoter (2-3kb)", "5' UTR", "3' UTR",
            "1st Exon", "Other Exon", "1st Intron", "Other Intron", "Downstream (<=300)", "Distal Intergenic")
percentages <- c(promoter_percent, promoter1kb_percent, promoter2kb_percent, fiveUTR_percent, threeUTR_percent,
                 first_exons_percent, exons_percent, first_introns_percent, introns_percent, downstream_percent, intergenic_percent)

# Ensure percentages are finite numbers
if (any(!is.finite(percentages))) {
  stop("One or more percentages are not finite numbers.")
}

# Define the colors
colors <- c("#a6cee3", "#1f78b4", "#b2df8a", "#33a02c", "#fb9a99",
            "#e31a1c", "#fdbf6f", "#ff7f00", "#cab2d6", "#6a3d9a", "#ffff99")

# Create the pie chart
par(mar=c(0,0,0,0))  # Add margins around the plot for better spacing
pie(1, radius=0.1, init.angle=0, col="white", border=NA, labels='')

floating.pie(-.5, 0, percentages, radius=1, startpos=0, col=colors, border="black")  # Adjust the radius here

# Create legend labels with percentages
legend_labels <- paste0(labels, " (", round(percentages, 2), "%)")

# Create the legend with the adjusted position
legend("topright", legend=legend_labels, fill=colors, cex=0.8, bty="n")
